const Layer = Fractale.create("Layer", {
    "pixels": [
        String
    ],
    "height": Number,
    "width": Number
});

const Frame = Fractale.create("Frame", {
    "layers": [
        {
            "$$__FractaleModel_TYPE_KEY": Layer,
            "$$__FractaleModel_OPTIONS_KEY": {
                "through": [
                    "height",
                    "width"
                ]
            }
        }
    ],
    "height": Number,
    "width": Number
});

const Sprite = Fractale.create("Sprite", {
    "frames": [
        {
            "$$__FractaleModel_TYPE_KEY": Frame,
            "$$__FractaleModel_OPTIONS_KEY": {
                "through": [
                    "height",
                    "width"
                ]
            }
        }
    ],
    "height": Number,
    "width": Number
});

const Character = Fractale.create("Character", {
    "name": String,
    "preview": Sprite,
    "dashes": [
        Sprite
    ],
    "stand": {
        "bottom": Sprite,
        "left": Sprite,
        "right": Sprite,
        "top": Sprite
    },
    "move": {
        "bottom": Sprite,
        "left": Sprite,
        "right": Sprite,
        "top": Sprite
    }
});

const Game = Fractale.create("Game", {
    "name": String,
    "type": String,
    "characters": [
        Character
    ]
});

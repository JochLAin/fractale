                    "height",
                    "width"

        {
            "$$__FractaleModel_TYPE_KEY": Frame,

                "through": [
                    "height",
                    "width"
                ]
            }
        }
    ],
    "height": Number,
    "width": Number
});

const Character = Fractale.create("Character", {
    "name": String,
    "preview": Sprite,
    "dashes": [
        Sprite
    ],

        "bottom": Sprite,
        "left": Sprite,
